[docs]class ContentEncodingProcessor(urllib2.BaseHandler):
    """A handler to add gzip capabilities to urllib2 requests """

    # add headers to requests
    def http_request(self, req):
        req.add_header("Accept-Encoding", "gzip, deflate")
        return req

    # decode
    def http_response(self, req, resp):
        old_resp = resp
        # gzip
        if resp.headers.get("content-encoding") == "gzip":
            gz = GzipFile(
                        fileobj=StringIO(resp.read()),
                        mode="r"
                      )
            resp = urllib2.addinfourl(gz, old_resp.headers, old_resp.url, old_resp.code)
            resp.msg = old_resp.msg
        # deflate
        if resp.headers.get("content-encoding") == "deflate":
            gz = StringIO(deflate(resp.read()))
            resp = urllib2.addinfourl(gz, old_resp.headers, old_resp.url, old_resp.code)  # 'class to add info() and
            resp.msg = old_resp.msg
        return resp

[docs]def get_html(url, fname):
    """download html content from the given urls"""

    try:
        encoding_support = ContentEncodingProcessor()
        opener = urllib2.build_opener(encoding_support, urllib2.HTTPHandler)
        content = opener.open(url)
        op = open(fname, 'wb')
        while 1:
            s = content.read()
            if not s:
                break
            op.write(s)
        op.close()
        content.close()
        return 1
    except Exception, e:
        if isinstance(e, urllib2.HTTPError):
            print 'http error: {0}'.format(e.code)
        elif isinstance(e, urllib2.URLError):
            print 'url error: socket timeout {0}'.format(e.__str__())
        else:
            print 'misc error: ' + e.__str__()




[docs]def get_url(lm_file, i=0):
    """get the baike URL of the landmarks in the given list"""

    urls = []
    p = {}
    # get names from file
    stuff = open(lm_file, 'r').read()
    lm_list = stuff.split('\n')[i:]
    # get urls
    for item in lm_list:
        p['word'] = item.decode('utf-8').encode('gb2312')
        item = urlencode(p)
        urls.append('http://baike.baidu.com/searchword/?' + item + '&pic=1&sug=1&enc=gbk')

    return urls
