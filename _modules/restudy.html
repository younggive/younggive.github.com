# reg sets
reg_card     = re.compile(r'(?<=sec-content0")([\s \S]*)(?=class="text_dir)')
reg_category = re.compile(r'(?<=/fenlei/)(?:.*?>)(.*?)(?=</a)')
reg_ID       = re.compile(r'(?<=\?lemmaid=).*?(?=")')
reg_PIC      = re.compile(r'(?<=card-image)(?:.*?)(http.*?)"')   # ??? PIC.group(0).split('src="')[1]
reg_SUM      = re.compile(r'(?<=card-summary-content"><p> ).*?(?=</p>)')
reg_NAME     = re.compile(r'(?<=h1)(?:.*>)(.*?)(?=<\/h1>)')
reg_INFO1    = re.compile(r'(?<=cardFirstTd">).*?(?=<\/td>)')
reg_INFO2    = re.compile(r'(?<=cardSecondTd">).*?(?=<\/td>)')
reg_rm_a     = re.compile(r'<a .*?>|<\/a>')     # reg <a></a> in a string
reg_rm_s     = re.compile(r'(^\s*)|(\s*$)')     # reg the spaces at the start & end of a string

[docs]def indent(elem, level=0):
    """Get pretty look"""

    i = "\n" + level * "  "
    if len(elem):
        if not elem.text or not elem.text.strip():
            elem.text = i + "  "
        for e in elem:
            indent(e, level + 1)
        if not e.tail or not e.tail.strip():
            e.tail = i
    if level and (not elem.tail or not elem.tail.strip()):
        elem.tail = i
    return elem



def process_value(x):
    if not x:
        return ''
    elif isinstance(x, list):
        x = [reg_rm_s.sub('', elem) for elem in x]
        return x
    else:
        if x.groups():
            return reg_rm_s.sub('', x.groups()[-1])
        else:
            return reg_rm_s.sub('', x.group())


def xml_form(card, root, d):
    """"""

    if d == 'lm':
        root = etree.SubElement(root, 'lm')
        if d in card.attr:
            for i in card.attr[d]:
                root.set(i, card.lable[i].decode('utf-8'))
    for x in card.tree[d]:
        if x[-1] == 's':
            count = 0
            for i in card.attr[x]:
                for j in card.lable[i]:
                    child = etree.SubElement(root, x)
                    child.set(i, j.decode('utf-8'))
                    child.text = card.lable['text'][count].decode('utf-8')
                    count += 1
        else:
            child = etree.SubElement(root, x)
            if card.lable[x]:
                child.text = card.lable[x].decode('utf-8')
            if x in card.tree:
                child = xml_form(card, child, x)

    return root
